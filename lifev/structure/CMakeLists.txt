
INCLUDE(TribitsPackageMacros)
INCLUDE(TribitsAddOptionAndDefine)
INCLUDE(TribitsLibraryMacros)

#
# A) Define the package
#

TRIBITS_PACKAGE(Structure)
SET(${PACKAGE_NAME}_VERSION "3.1")

#
# B) Set up package-specific options
#

TRIBITS_ADD_SHOW_DEPRECATED_WARNINGS_OPTION()

TRIBITS_ADD_OPTION_AND_DEFINE(LifeV_${PACKAGE_NAME}_ENABLE_ANISOTROPIC
  ENABLE_ANISOTROPIC_LAW
  "Enable Anisotropic"
  OFF )

TRIBITS_ADD_OPTION_AND_DEFINE(LifeV_${PACKAGE_NAME}_COLORING_MESH
  COLORING_MESH
  "Enable Coloring Mesh"
  OFF )

TRIBITS_ADD_OPTION_AND_DEFINE(LifeV_${PACKAGE_NAME}_COMPUTATION_JACOBIAN
  COMPUTATION_JACOBIAN
  "Enable Computation Jacobian"
  OFF )

TRIBITS_ADD_OPTION_AND_DEFINE(LifeV_${PACKAGE_NAME}_EXPORTVECTORS
  EXPORTVECTORS
  "Enable Vectors Export for Debug"
  OFF )

IF(NOT ENABLE_ANISOTROPIC_LAW)
  MESSAGE(STATUS "LifeV_Structure: only isotropic laws available")
ENDIF()

TRIBITS_CONFIGURE_FILE(${PACKAGE_NAME}_config.h)

INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR})

SET(HEADERS "")
SET(SOURCES "")

ADD_SUBDIRECTORY(solver)
SET(HEADERS ${HEADERS} ${solver_HEADERS})
SET(SOURCES ${SOURCES} ${solver_SOURCES})

ADD_SUBDIRECTORY(solver/isotropic)
SET(HEADERS ${HEADERS} ${solverIsotropic_HEADERS})
SET(SOURCES ${SOURCES} ${solverIsotropic_SOURCES})

ADD_SUBDIRECTORY(fem)
SET(HEADERS ${HEADERS} ${fem_HEADERS})
SET(SOURCES ${SOURCES} ${fem_SOURCES})

IF(ENABLE_ANISOTROPIC_LAW)
  ADD_SUBDIRECTORY(solver/anisotropic)
  SET(HEADERS ${HEADERS} ${solverAnisotropic_HEADERS})
  SET(SOURCES ${SOURCES} ${solverAnisotropic_SOURCES})
ENDIF()

TRIBITS_EXCLUDE_AUTOTOOLS_FILES()
# This is the list of files excluded from the tarball
TRIBITS_EXCLUDE_FILES()

# Add library target
TRIBITS_ADD_LIBRARY(
  lifevstructure
  HEADERS ${HEADERS}
  SOURCES ${SOURCES}
)

TRIBITS_ADD_TEST_DIRECTORIES(testsuite)
TRIBITS_ADD_EXAMPLE_DIRECTORIES(examples)

IF(PDFLATEX_COMPILER)
  ADD_SUBDIRECTORY(doc)
ENDIF()

#
# Do standard postprocessing
#

TRIBITS_PACKAGE_POSTPROCESS()
