
INCLUDE(TribitsPackageMacros)
INCLUDE(TribitsAddOptionAndDefine)
INCLUDE(TribitsLibraryMacros)

#
# A) Define the package
#

TRIBITS_PACKAGE(ETA)
SET(${PACKAGE_NAME}_VERSION "0.1")

#
# B) Set up package-specific options
#

TRIBITS_ADD_SHOW_DEPRECATED_WARNINGS_OPTION()

INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR})

SET(HEADERS "")
SET(SOURCES "")

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})

ADD_SUBDIRECTORY(fem)
SET(HEADERS ${HEADERS} ${fem_HEADERS})
SET(SOURCES ${SOURCES} ${fem_SOURCES})

ADD_SUBDIRECTORY(expression)
SET(HEADERS ${HEADERS} ${expression_HEADERS})
SET(SOURCES ${SOURCES} ${expression_SOURCES})

ADD_SUBDIRECTORY(array)
SET(HEADERS ${HEADERS} ${array_HEADERS})
SET(SOURCES ${SOURCES} ${array_SOURCES})

TRIBITS_EXCLUDE_AUTOTOOLS_FILES()
# This is the list of files excluded from the tarball
TRIBITS_EXCLUDE_FILES()

# Add library target
TRIBITS_ADD_LIBRARY(
  lifeveta
  HEADERS ${HEADERS}
  SOURCES ${SOURCES}
)

TRIBITS_ADD_TEST_DIRECTORIES(tutorials)
TRIBITS_ADD_TEST_DIRECTORIES(testsuite)

#
# Do standard postprocessing
#

TRIBITS_PACKAGE_POSTPROCESS()
